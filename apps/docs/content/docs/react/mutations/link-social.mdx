---
title: Link Social
description: Link a social provider to the current user.
---

Kicks off the OAuth redirect — the linked account appears in `useListAccounts` after the provider redirects back.

## Usage

```tsx
import { useLinkSocial } from "@better-auth-ui/react"

const { mutate: linkSocial } = useLinkSocial()

linkSocial({
  provider: "github",
  callbackURL: "/account"
})
```

## Options factory

```tsx
import { linkSocialOptions } from "@better-auth-ui/react"
import { useMutation } from "@tanstack/react-query"

const { mutate } = useMutation(linkSocialOptions(authClient))
```
