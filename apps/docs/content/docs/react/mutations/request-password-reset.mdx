---
title: Request Password Reset
description: Send a password reset email for the forgot-password flow.
---

## Usage

```tsx
import { useRequestPasswordReset } from "@better-auth-ui/react"

const { mutate: requestPasswordReset } = useRequestPasswordReset()

requestPasswordReset({
  email: "alice@example.com",
  redirectTo: "/auth/reset-password"
})
```

## Options factory

```tsx
import { requestPasswordResetOptions } from "@better-auth-ui/react"
import { useMutation } from "@tanstack/react-query"

const { mutate } = useMutation(requestPasswordResetOptions(authClient))
```
