---
title: useSignUpEmail
description: Email/password sign-up mutation.
---

Resets the session query on success so the new session is refetched.

## Usage

```tsx
import { useSignUpEmail } from "@better-auth-ui/react"

const { mutate: signUpEmail } = useSignUpEmail()

signUpEmail({
  email: "alice@example.com",
  password: "hunter2",
  name: "Alice"
})
```

## Options factory

```tsx
import { signUpEmailOptions } from "@better-auth-ui/react"
import { useMutation } from "@tanstack/react-query"

const { mutate } = useMutation(signUpEmailOptions(authClient))
```
