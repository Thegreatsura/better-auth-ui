---
title: Reset Password
description: Complete the password reset flow with a token.
---

## Usage

```tsx
import { useResetPassword } from "@better-auth-ui/react"

const { mutate: resetPassword } = useResetPassword()

resetPassword({
  newPassword: "hunter3",
  token
})
```

## Options factory

```tsx
import { resetPasswordOptions } from "@better-auth-ui/react"
import { useMutation } from "@tanstack/react-query"

const { mutate } = useMutation(resetPasswordOptions(authClient))
```
